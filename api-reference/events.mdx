## Endpoint

```
GET /events/:id
```

Retrieves comprehensive details about a single event, including title, description, location, timing, and organizer information.

## Path Parameters

<ParamField path="id" type="string" required>
  The unique event identifier (format: `evt_xxxxx`)
  
  Example: `evt_abc123def456`
</ParamField>

## Request Example

<RequestExample>

```bash cURL
curl https://api.evento.so/public/v1/events/evt_abc123def456 \
  -H "Authorization: Bearer evento_xxxxxxxxxxxxxxxxxxxxxxxx"
```

```javascript JavaScript
const eventId = 'evt_abc123def456';

const response = await fetch(
  `https://api.evento.so/public/v1/events/${eventId}`,
  {
    headers: {
      'Authorization': `Bearer ${process.env.EVENTO_API_KEY}`
    }
  }
);

const { data } = await response.json();
console.log(data.event);
```

```python Python
import os
import requests

event_id = 'evt_abc123def456'
headers = {
    'Authorization': f'Bearer {os.environ["EVENTO_API_KEY"]}'
}

response = requests.get(
    f'https://api.evento.so/public/v1/events/{event_id}',
    headers=headers
)

data = response.json()
print(data['data']['event'])
```

</RequestExample>

## Response Fields

<ResponseField name="event" type="object">
  The event object containing all event details

  <Expandable title="Event Properties">
    <ResponseField name="id" type="string">
      Unique event identifier
    </ResponseField>

    <ResponseField name="title" type="string">
      Event title
    </ResponseField>

    <ResponseField name="description" type="string">
      Full event description (may contain markdown)
    </ResponseField>

    <ResponseField name="slug" type="string">
      URL-friendly event identifier
    </ResponseField>

    <ResponseField name="startDate" type="string">
      Event start time (ISO 8601 format)
    </ResponseField>

    <ResponseField name="endDate" type="string">
      Event end time (ISO 8601 format)
    </ResponseField>

    <ResponseField name="timezone" type="string">
      IANA timezone identifier
    </ResponseField>

    <ResponseField name="location" type="object">
      Event location details
      
      <Expandable title="Location Properties">
        <ResponseField name="type" type="string">
          Location type: `physical`, `virtual`, or `hybrid`
        </ResponseField>

        <ResponseField name="name" type="string">
          Venue or platform name
        </ResponseField>

        <ResponseField name="address" type="string">
          Physical address (if applicable)
        </ResponseField>

        <ResponseField name="coordinates" type="object">
          Geographic coordinates (if applicable)
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="coverImage" type="string">
      URL to event cover image
    </ResponseField>

    <ResponseField name="capacity" type="number">
      Maximum number of attendees
    </ResponseField>

    <ResponseField name="ticketsSold" type="number">
      Number of tickets sold
    </ResponseField>

    <ResponseField name="ticketsAvailable" type="number">
      Number of tickets remaining
    </ResponseField>

    <ResponseField name="price" type="object">
      Ticket pricing information
      
      <Expandable title="Price Properties">
        <ResponseField name="amount" type="number">
          Price in smallest unit (sats)
        </ResponseField>

        <ResponseField name="currency" type="string">
          Currency type (typically `sats`)
        </ResponseField>

        <ResponseField name="formatted" type="string">
          Human-readable price string
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="organizer" type="object">
      Event organizer information
      
      <Expandable title="Organizer Properties">
        <ResponseField name="id" type="string">
          Organizer user ID
        </ResponseField>

        <ResponseField name="username" type="string">
          Organizer username
        </ResponseField>

        <ResponseField name="displayName" type="string">
          Organizer display name
        </ResponseField>

        <ResponseField name="avatar" type="string">
          URL to organizer avatar
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="tags" type="array">
      Array of event tags/categories
    </ResponseField>

    <ResponseField name="isPublic" type="boolean">
      Whether the event is publicly visible
    </ResponseField>

    <ResponseField name="requiresApproval" type="boolean">
      Whether ticket purchases require organizer approval
    </ResponseField>

    <ResponseField name="createdAt" type="string">
      Event creation timestamp (ISO 8601)
    </ResponseField>

    <ResponseField name="updatedAt" type="string">
      Last update timestamp (ISO 8601)
    </ResponseField>
  </Expandable>
</ResponseField>
